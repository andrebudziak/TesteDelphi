CREATE SCHEMA `TESTE` ;

CREATE TABLE `TESTE`.`CLIENTE` (
  `CODIGO` INT NOT NULL AUTO_INCREMENT,
  `NOME` VARCHAR(100) NOT NULL,
  `CIDADE` VARCHAR(100) NULL,
  `UF` VARCHAR(2) NULL,
  PRIMARY KEY (`CODIGO`));

CREATE TABLE `TESTE`.`PRODUTO` (
  `CODIGO` INT NOT NULL AUTO_INCREMENT,
  `DESCRICAO` VARCHAR(100) NOT NULL,
  `PRECOVENDA` DECIMAL NOT NULL,
  PRIMARY KEY (`CODIGO`));


CREATE TABLE `TESTE`.`PEDIDO` (
  `CODIGO` INT NOT NULL AUTO_INCREMENT,
  `CODIGO_CLIENTE` INT NOT NULL,
  `DATA_EMISSAO` DATE NOT NULL,
  `VALOR_TOTAL` DECIMAL NOT NULL,
  `NUMERO` INT NOT NULL,
  PRIMARY KEY (`CODIGO`),
  FOREIGN KEY (`CODIGO_CLIENTE`) REFERENCES CLIENTE(`CODIGO`));
  
CREATE TABLE `TESTE`.`PEDIDO_ITEM` (
  `CODIGO` INT NOT NULL AUTO_INCREMENT,
  `CODIGO_PEDIDO` INT NOT NULL,
  `CODIGO_PRODUTO` INT NOT NULL,
  `VALOR` DECIMAL NOT NULL,
  `QTD` INT NOT NULL,
  PRIMARY KEY (`CODIGO`),
  FOREIGN KEY (`CODIGO_PRODUTO`) REFERENCES PRODUTO(`CODIGO`),
  FOREIGN KEY (`CODIGO_PEDIDO`) REFERENCES PEDIDO(`CODIGO`) ON DELETE CASCADE);
